397 inline void TYPixelsOverlapRemove(TY_PIXEL_DESC* lut, uint32_t count, uint32_t imageW, uint32_t imageH)
398 {
399  uint16_t* mappedDepth = (uint16_t*)calloc(imageW*imageH, sizeof(uint16_t));
400  for(size_t i = 0; i < count; i++) {
401  if(lut[i].x < 0 || lut[i].y < 0 || lut[i].x >= imageW || lut[i].y >= imageH) continue;
402  uint32_t offset = lut[i].y * imageW + lut[i].x;
403  if(lut[i].depth && (mappedDepth[offset] == 0 || mappedDepth[offset] >= lut[i].depth))
404  mappedDepth[offset] = lut[i].depth;
405  }
406  TYDepthImageFillEmptyRegion(mappedDepth, imageW, imageH);
407  for(size_t i = 0; i < count; i++) {
408  if(lut[i].x < 0 || lut[i].y < 0 || lut[i].x >= imageW || lut[i].y >= imageH) {
409  continue;
410  } else {
411  uint32_t offset = lut[i].y * imageW + lut[i].x;
412  int32_t delt = lut[i].depth - mappedDepth[offset];
413  if(lut[i].depth && delt > 10) {
414  lut[i].x = -1;
415  lut[i].y = -1;
416  lut[i].depth = 0;
417  }
418  }
419  }
420  free(mappedDepth);
421 }
